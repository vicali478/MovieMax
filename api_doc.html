Search & explore

Search endpoint supports title, type and page.

GET
/explore/search

Query params:

  • title (required)
  • page (optional)
  • type (optional) â€” movie or tv

{
    "success":true,
    "page":1,
    "totalPages":3,
    "totalResults":35,
    "results":[ ... ]
}
Example usage 
async function searchTitle(title: string, page = 1, type?: 'movie'|'tv', apiKey?: string) {
    const params = new URLSearchParams({ title, page: String(page) });
    if (type) params.set('type', type);
    const res = await fetch(`https://api.wustream.com/explore/search?${params.toString()}`, {
        headers: { "x-api-key": apiKey || 'YOUR_KEY', "Accept": "application/json" }
    });
    if (!res.ok) throw new Error(await res.text());
    return res.json();
}

Additional notes & best practices

  • Always use HTTPS.
  • Validate API key server-side and enforce rate limits.
  • Filtered results typically exclude items missing posterUrl.
  • For large downloads, the server may return a 302 redirect to a CDN; follow redirects.
  • Use Accept-Language for localization where supported.
  • Cache list endpoints (e.g., trending) for a short period to reduce load.

Flutter (Dart) â€” simple widget to fetch a movie by id

import 'package:flutter/material.dart';
import 'package:http/http.dart' as http;
import 'dart:convert';

class Movie {
  final int id;
  final String title;
  final String posterUrl;
  Movie({required this.id, required this.title, required this.posterUrl});
  factory Movie.fromJson(Map<String,dynamic> j) => Movie(
    id: j['id'],
    title: j['title'] ?? '',
    posterUrl: j['posterUrl'] ?? '',
  );
}

Future<Movie> fetchMovie(int id, String apiKey, {String base = 'https://api.wustream.com'}) async {
  final url = Uri.parse('$base/movies/id/$id');
  final res = await http.get(url, headers: {
    'x-api-key': apiKey,
    'Accept': 'application/json',
  });
  if (res.statusCode == 200) {
    return Movie.fromJson(json.decode(res.body));
  }
  throw Exception('Failed to load movie: ${res.statusCode}');
}

class MovieByIdWidget extends StatefulWidget {
  final int id;
  final String apiKey;
  const MovieByIdWidget({required this.id, required this.apiKey, Key? key}): super(key:key);
  @override _MovieByIdWidgetState createState() => _MovieByIdWidgetState();
}

class _MovieByIdWidgetState extends State<MovieByIdWidget> {
  late Future<Movie> movie;
  @override
  void initState() {
    super.initState();
    movie = fetchMovie(widget.id, widget.apiKey);
  }
  @override
  Widget build(BuildContext context) {
    return FutureBuilder<Movie>(
      future: movie,
      builder: (context, snapshot) {
        if (snapshot.connectionState != ConnectionState.done) return Center(child:CircularProgressIndicator());
        if (snapshot.hasError) return Text('Error: ${snapshot.error}');
        final m = snapshot.data!;
        return ListTile(
          leading: m.posterUrl.isNotEmpty ? Image.network(m.posterUrl, width:56, fit:BoxFit.cover) : null,
          title: Text(m.title),
          subtitle: Text('ID: ${m.id}'),
        );
      },
    );
  }
}

Vanilla JavaScript â€” small helper and usage

async function apiFetch(path, apiKey='YOUR_KEY', base='https://api.wustream.com') {
  const url = path.startsWith('http') ? path : `${base.replace(/\/+$/,'')}/${path.replace(/^\/+/,'')}`;
  const res = await fetch(url, { headers: { 'x-api-key': apiKey, 'Accept': 'application/json' }});
  if (!res.ok) throw new Error(`${res.status} ${await res.text()}`);
  return res.json();
}

// Usage
apiFetch('/movies/trending/1', 'YOUR_KEY')

React (function component) â€” fetch and render trending movies

// Example React component (use in CRA / Vite)
import React, { useEffect, useState } from 'react';

export default function TrendingMovies({ apiKey = process.env.REACT_APP_API_KEY, base = process.env.REACT_APP_BASE_URL || 'https://api.wustream.com' }) {
  const [movies, setMovies] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    let mounted = true;
    (async () => {
      try {
        const res = await fetch(`${base.replace(/\/+$/,'')}/movies/trending/1`, {
          headers: { 'x-api-key': apiKey || 'YOUR_KEY', 'Accept': 'application/json' }
        });
        if (!res.ok) throw new Error(await res.text());
        const json = await res.json();
        if (mounted) setMovies(json.movies || []);
      } catch (e) {
        if (mounted) setError(e.message || 'Error');
      } finally {
        if (mounted) setLoading(false);
      }
    })();
    return () => { mounted = false; };
  }, [apiKey, base]);

  if (loading) return <div>Loading...</div>;
  if (error) return <div style={{color:'salmon'}}>{error}</div>;

  return <div style={{display:'grid',gridTemplateColumns:'repeat(auto-fill,minmax(140px,1fr))',gap:12}}>
    {movies.map(m => (
      <div key={m.id} style={{background:'#071623',padding:8,borderRadius:8}}>
        <img src={m.posterUrl} alt={m.title} style={{width:'100%',borderRadius:6}} />
        <div style={{marginTop:8,fontSize:13,fontWeight:700}}>{m.title}</div>
        <div style={{fontSize:12,color:'#94a3b8'}}>{m.releaseDate || ''}</div>
      </div>
    ))}
  </div>;
}

Hints:

  • Store API keys server-side or in environment variables. Do not expose production keys in client bundles.
  • For production React apps, use environment variables (REACT_APP_... / VITE_...).
  • Follow redirects when downloading binary content (curl -L) or let the browser handle them for fetch requests.
